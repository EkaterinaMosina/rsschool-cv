function pigIt(str){
return str.split(' ').map(function(word){
return (word.toUpperCase()===word.toLowerCase()) ? word : (word.substr(1) + word.charAt(0) + 'ay');
}).join(' ');
};

var maxSequence = function(arr){
let maxSum=0;
let interSum=0;
for (let item of arr) {
interSum+=item;
maxSum = Math.max(maxSum, interSum);
if (interSum <0) interSum = 0;
}
return maxSum;
}
